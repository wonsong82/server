UBUNTU:
apt install ocaml
apt install make
mkdir /etc/unison
wget <source>.tar.gz
cd /etc/unison/unison-<version>/src
make UISTYLE=text
./unison <- to test
./unison-fsmonitor <- to test
mv unison unison-fsmonitor /usr/local/bin
